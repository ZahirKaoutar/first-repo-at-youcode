Mini projet #include <stdio.h>


void afficherNotes(int notes[], int taille) {
    printf("Les notes sont : ");
    for (int i = 0; i < taille; i++) {
        printf("%d ", notes[i]);
    }
    printf("\n");
}

float calculerMoyenne(int notes[], int taille) {
    int somme = 0;
    for (int i = 0; i < taille; i++) {
        somme += notes[i];
    }
    return (float)somme / taille;
}


int trouverMax(int notes[], int taille) {
    int max = notes[0];  
    for (int i = 1; i < taille; i++) {
        if (notes[i] > max) {
            max = notes[i]; 
        }
    }
    return max;
}

int main() {
    int taille;

    printf("Entrez le nombre de notes : ");
    scanf("%d", &taille);

  
    if (taille <= 0) {
        printf("Le nombre de notes doit être positif.\n");
        return 1;
    }

  
    int notes[taille];

    
    for (int i = 0; i < taille; i++) {
        printf("Entrez la note %d : ", i + 1);
        scanf("%d", &notes[i]);
    }

    afficherNotes(notes, taille);

    float moyenne = calculerMoyenne(notes, taille);
    printf("La moyenne des notes est : %.2f\n", moyenne);

    int max = trouverMax(notes, taille);
    printf("La note maximale est : %d\n", max);

    return 0;
}
